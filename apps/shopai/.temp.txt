// Generate hourly leads for today (24 hours) as { [date: string]: HourlyLead[] }
const generateTodayHourlyLeads = (): { [date: string]: HourlyLead[] } => {
    const now = new Date()
    const hours: HourlyLead[] = []

    // Set to today's midnight for date key
    const dateKey = now.toISOString().slice(0, 10) // "yyyy-mm-dd"

    for (let hour = 0; hour < 24; hour++) {
        const hourDate = new Date(now)
        hourDate.setHours(hour, 0, 0, 0)
        // Generate realistic lead counts (higher during business hours)
        const baseLeads = hour >= 9 && hour <= 17 ? 15 : 5
        const randomVariation = Math.floor(Math.random() * 10)
        const leads = Math.max(0, baseLeads + randomVariation)

        const time = hourDate.toLocaleTimeString("en-US", {
            hour: "numeric",
            minute: "2-digit",
            hour12: true,
        })

        hours.push({
            hour,
            leads,
            time,
        })
    }

    return {
        [dateKey]: hours,
    }
}
